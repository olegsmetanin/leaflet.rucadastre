Usage

Add script

<script src="http://olegsmith.github.com/leaflet.rucadastre/leaflet.rucadastre.js"></script>

Add RuCadastre layer

<script>
    var map = L.map('map').setView([55.751667, 37.617778], 13);
    L.control.scale().addTo(map);
    L.tileLayer('http://{s}.tile.cloudmade.com/BC9A493B41014CAABB98F0471D759707/997/256/{z}/{x}/{y}.png', {
         maxZoom: 18
        , attribution: 'Map data &copy; <a href="http://openstreetmap.org">OpenStreetMap</a> contributors, <a href="http://creativecommons.org/licenses/by-sa/2.0/">CC-BY-SA</a>, Imagery � <a href="http://cloudmade.com">CloudMade</a>'
        }).addTo(map);
    var rucadastre = new L.RuCadastre();
    map.addLayer(rucadastre);
</script>

Add RuCadastre identify tool

<script src="https://raw.github.com/calvinmetcalf/leaflet-ajax/master/src/ajax.js"></script> 

<script>
    var ruCadastreIdentify = new L.RuCadastreIdentify();
    map.addLayer(ruCadastreIdentify);
</script>
